# docker 사용법

```
docker pull nginx
docker image ls

docker create nginx

docker ps           # alive processor
docker ps -a        # stop/live processor
docker run nginx    # fore ground
docker run -d --name my-seb-werver nginx    # back ground
docker stop  981
docker rm 981

docker image ls
docker image rm 911
docker image rm nginx

docker run -d nginx
docker run -d -p 4000:80 nginx
docker run -d -p 80:80 nginx

docker stop 643     # safe 종료
docker kill 643     # 강제 종료
docker rm $(docker ps -qa)  # 중지되어 있는 모든 컨테이너 삭제
docker rm -f 786    # 실행주인 컨테이너 삭제

docker logs 789     # 컨테이너 로그 확인
docker logs --tail 10 ee6
dokcer logs -f 2ed  # 실시간 로그 볼 때
docker logs --tail 0 -f 6cf # 실시간 로그

docker exec -it 433 bash    # 컨테이너 내부로
 ----- 컨테이너 내부 ----
exit        # 컨테이너 외부로 빠져 나올 때
```

```
docker run -d mysql
docker run -d -p 3306:3306 mysql
lsof -i:3306    # 포트 상태 확인
kill -9 <pid>

docker run -d -e MYSQL_ROOT_PASSWORD=passwd123 -p 3306:3306 mysql
docker exec -it 43e bash
echo $MYSQL_ROOT_PASSWORD       # 환경 변수 확인
passwd123
exit
```
